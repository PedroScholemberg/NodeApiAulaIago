CREATE DATABASE APICRUD;
USE APICRUD;

CREATE TABLE users(
id INT(5) NOT NULL primary KEY,
nome VARCHAR(45),
data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE images(
id INT(5) NOT NULL primary KEY,
referencia VARCHAR(45),
data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
titulo VARCHAR(45)
);

select * from users;
